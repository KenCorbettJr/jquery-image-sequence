Jquery Image Sequence

Basic usage

          $(function () {
            $(".basic-usage").imagesequence({
              src: 'anim_a.jpg',
              width: 1280,
              height: 720
            });
          });

Rewind

          $(function () {
            $(".rewind").imagesequence({
              src: 'anim_a.jpg',
              width: 1280,
              height: 720,
              rewind: true
            });
          });

Controls

          $(function () {
            var imagesequence = $(".controls").imagesequence({
              src: 'anim_a.jpg',
              width: 1280,
              height: 720
            });
            $('.play').on('click', function(){
              imagesequence.play();
            });
            $('.pause').on('click', function(){
              imagesequence.pause();
            });
            $('.rewind-button').on('click', function(){
              imagesequence.rewind();
            });
          });
